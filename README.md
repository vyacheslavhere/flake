# Flake
❄️ A statically typed, functional, friendly language for WASM.

# Example
```flake
import std::io

type Cat {
  fish: i16,
  health: f32,
}
impl {
  fn get_fish() -> i16 {
    return cat.fish 
  }
  fn get_health() -> f32 {
    return cat.health
  }
}

fn main() {
  io.println(cat.get_fish().to_string())
}
```
